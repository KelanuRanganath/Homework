\subsection*{Part a}
Wts.
\eqe{
mu(A union B) &= mu(A) + mu(B) - mu(A intersect B)
}
\subsection*{Solution}
\eqe{
mu(A) &= mu(A minus B union (A intersect B))
&= mu(A minus B) + mu(A intersect B)
}
\eqe{
mu(A union B) &= mu((A minus B) union (B minus A) union (A intersect B))
&= mu(A minus B) + mu(B minus A) + mu(A intersect B)
}

Combining equations (1.2) and (1.3) we get
\eqe{
\mu(A) + \mu(B) &= mu(A minus B) + mu(B minus A) + 2mu(A intersect B)
&= mu(A union B) + mu(A intersect B)
}

So it follows that
\eqe{
mu(A union B) &= mu(A) + mu(B) - mu(A intersect B)
}
\subsection*{Part b}
Wts.
\eqe{
B subset A implies mu(A intersect B^c) &= mu(A) - mu(B)
}
\subsection*{Solution}
\eqe{
mu(A) &= mu(A minus B union B)
&= mu(A minus B) + mu(B)
&= mu(A intersect B^c) + mu(B)
}

So it follows that
\eqe{
mu(A) - mu(B) &= mu(A intersect B^c)
}

\subsection*{Part a}
Wts. $f$ and $g$ are not measurable.
\subsection*{Solution}
$f$ and $g$ are measurable iff for every interval $I \subseteq \mathbb{R}$ the pullback $f^{-1}(I) \in \mathcal{F}$.
\eqe{
f^{-1}([0,1]) &= g^{-1}([1,2]) = [0,1] \notin \mathcal{F}
}
So $f$ and $g$ are not measurable.
